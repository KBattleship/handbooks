        int[] arr = new int[]{10,1,0,9,22,77,12,883,99983,2772,848,3663,2626,737,2772,8388,266,83,72,7272,83883,27727,263,840,2740,884};

    public static void print2(int[] arr) {
        int k = 10;
        int[] vec = new int[k];
        // ÈáçÂÜôPriorityQueue‰∏∫Â§ßÊ†πÂ†Ü
        // PriorityQueue<Integer> queue = new PriorityQueue<>();ÈªòËÆ§‰∏∫Â∞èÊ†πÂ†Ü
        PriorityQueue<Integer> queue = new PriorityQueue<>((num1, num2) -> num2 - num1);
        for (int i = 0; i < k; ++i) {
            queue.offer(arr[i]);
        }
        for (int i = k; i < arr.length; ++i) {
            if (queue.peek() > arr[i]) {
                queue.poll();
                queue.offer(arr[i]);
            }
        }
        for (int i = 0; i < k; ++i) {
            vec[i] = queue.poll();
        }
        System.out.println(Arrays.toString(vec));
    }
}

class Lock{
    private int flag = 0;

    public int getFlag() {
        return flag;
    }

    public void setFlag(int flag) {
        this.flag = flag;

class A extends Thread{
    private Lock lock;

    public A(Lock lock) {
        this.lock = lock;
    }

    @Override
    public void run() {
        synchronized (lock) {
            try {
                System.out.println("thread A starting ");
                if (lock.getFlag() > 0) {
                    lock.notify();
                } else {
                    lock.setFlag(1);
                }
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
}

class B extends Thread{
    private Lock lockA;
    private Lock lockB;

    public B(Lock lockA,Lock lockB) {
        this.lockA = lockA;
        this.lockB = lockB;
    }

    @Override
    public void run() {
        synchronized (lockA) {

            try {
                if (lockA.getFlag() == 0) {
                    lockA.setFlag(1);
                    lockA.wait();
                }
                System.out.println("Thread B starting");

                synchronized (lockB) {
                    if (lockB.getFlag() > 0) {
                        lockB.notify();
                    } else {
                        lockB.setFlag(1);
                    }
                }
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
}
